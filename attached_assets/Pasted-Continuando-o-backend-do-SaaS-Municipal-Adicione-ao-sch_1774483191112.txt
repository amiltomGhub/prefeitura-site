// ─── NEWS — Notícias e comunicados ───────────

model News {
  id              String        @id @default(uuid()) @db.Uuid
  title           String        @db.VarChar(255)
  slug            String        @unique @db.VarChar(300)
  excerpt         String?       @db.Text          // Resumo (gerado ou manual)
  content         String        @db.Text          // HTML do TipTap
  coverImageUrl   String?       @db.Text
  coverImageAlt   String?       @db.VarChar(255)  // Obrigatório se tiver imagem
  status          ContentStatus @default(DRAFT)
  isFeatured      Boolean       @default(false)
  categoryId      String?       @db.Uuid
  category        NewsCategory_Model? @relation(fields: [categoryId], references: [id])
  secretariaId    String?       @db.Uuid
  secretaria      Secretaria?   @relation(fields: [secretariaId], references: [id])
  authorId        String        @db.Uuid
  author          User          @relation(fields: [authorId], references: [id])
  tags            String[]
  viewCount       Int           @default(0)
  // SEO
  metaTitle       String?       @db.VarChar(70)
  metaDescription String?       @db.VarChar(160)
  ogImageUrl      String?       @db.Text
  publishedAt     DateTime?     @db.Timestamptz   // null = rascunho
  scheduledAt     DateTime?     @db.Timestamptz   // para publicação agendada
  createdAt       DateTime      @default(now()) @db.Timestamptz
  updatedAt       DateTime      @updatedAt @db.Timestamptz
  deletedAt       DateTime?     @db.Timestamptz

  versions        NewsVersion[]

  @@index([slug])
  @@index([status])
  @@index([isFeatured])
  @@index([publishedAt])
  @@index([categoryId])
  @@map("news")
  @@schema("TENANT_SCHEMA_PLACEHOLDER")
}

model NewsVersion {
  id        String   @id @default(uuid()) @db.Uuid
  newsId    String   @db.Uuid
  news      News     @relation(fields: [newsId], references: [id], onDelete: Cascade)
  content   String   @db.Text
  savedById String   @db.Uuid
  savedBy   User     @relation(fields: [savedById], references: [id])
  createdAt DateTime @default(now()) @db.Timestamptz
  @@map("news_versions")
  @@schema("TENANT_SCHEMA_PLACEHOLDER")
}

model NewsCategory_Model {
  id        String @id @default(uuid()) @db.Uuid
  name      String @db.VarChar(100)
  slug      String @unique @db.VarChar(100)
  color     String? @db.VarChar(7)
  news      News[]
  @@map("news_categories")
  @@schema("TENANT_SCHEMA_PLACEHOLDER")
}

// ─── PAGES — Páginas estáticas ───────────────

model Page {
  id          String        @id @default(uuid()) @db.Uuid
  title       String        @db.VarChar(255)
  slug        String        @unique @db.VarChar(300)
  status      ContentStatus @default(DRAFT)
  isProtected Boolean       @default(false)  // true = não pode ser deletada
  authorId    String        @db.Uuid

  metaTitle   String?       @db.VarChar(70)
  metaDescription String?   @db.VarChar(160)
  createdAt   DateTime      @default(now()) @db.Timestamptz
  updatedAt   DateTime      @updatedAt @db.Timestamptz

  blocks      PageBlock[]

  @@map("pages")
  @@schema("TENANT_SCHEMA_PLACEHOLDER")
}

  sortOrder Int           @default(0)

  @@index([pageId])
  @@map("page_blocks")
  @@schema("TENANT_SCHEMA_PLACEHOLDER")
}

// ─── LEGISLATION — Atos normativos ───────────

model Legislation {
  id          String          @id @default(uuid()) @db.Uuid
  type        LegislationType
  number      String          @db.VarChar(50)   // "Decreto nº 1.234"
  year        Int
  summary     String          @db.Text          // Ementa
  content     String?         @db.Text          // Texto completo (opcional)
  fileUrl     String?         @db.Text          // PDF
  fileName    String?         @db.VarChar(255)
  status      ContentStatus   @default(PUBLISHED)

  revokedAt   DateTime?       @db.Timestamptz   // Se revogado
  revokedById String?         @db.Uuid          // Revogado pelo ato ID
  tags        String[]
  viewCount   Int             @default(0)
  downloadCount Int           @default(0)
  createdById String          @db.Uuid
  createdBy   User            @relation(fields: [createdById], references: [id])
  createdAt   DateTime        @default(now()) @db.Timestamptz
  updatedAt   DateTime        @updatedAt @db.Timestamptz

  @@unique([type, number, year])
  @@index([type])
  @@index([year])
  @@index([publishedAt])
  @@map("legislation")
  @@schema("TENANT_SCHEMA_PLACEHOLDER")
}

  openingDate     DateTime     @db.Timestamptz    // Data de abertura
  closingDate     DateTime?    @db.Timestamptz    // Data de encerramento

  resultUrl       String?      @db.Text           // PDF do resultado
